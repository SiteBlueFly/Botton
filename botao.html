    <script> 

        function createButterflies() { 

            const button = document.getElementById('cta-metamorfose'); 

            const buttonRect = button.getBoundingClientRect(); 

             

            for (let i = 0; i < 10; i++) { 

                const butterfly = document.createElement('div'); 

                butterfly.innerHTML = '🦋'; 

                butterfly.className = 'butterfly'; 

                 

                const randomX = Math.random() * buttonRect.width; 

                const randomY = Math.random() * buttonRect.height; 

                 

                butterfly.style.left = (buttonRect.left + randomX) + 'px'; 

                butterfly.style.top = (buttonRect.top + randomY) + 'px'; 

                butterfly.style.animationDuration = (Math.random() * 1.5 + 0.5) + 's'; 

                 

                document.body.appendChild(butterfly); 

                 

                setTimeout(() => butterfly.remove(), 2000); 

            } 

        } 


        // Remove borboletas quando o mouse sai 

        document.getElementById('cta-metamorfose').addEventListener('mouseleave', () => { 

            document.querySelectorAll('.butterfly').forEach(el => el.remove()); 

        }); 


        window.addEventListener('resize', () => { 

            document.querySelectorAll('.butterfly').forEach(el => el.remove()); 

        }); 

    </script>
